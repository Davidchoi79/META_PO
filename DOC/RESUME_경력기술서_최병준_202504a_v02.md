        C1 --> C2 --> C3 --> C4 --> C5 --> C6 --> C7
        C7 -.피드백 루프.-> C4
    end
  
    %% 인터페이스 요소
    UI1["위젯 관리자<br>대시보드"]
    UI2["위젯 설정<br>인터페이스"]
    UI3["고객사 웹사이트<br>적용 예시"]
  
    C4 -.-> UI2
    C7 -.-> UI1
    C6 -.-> UI3
  
    %% 스타일 정의
